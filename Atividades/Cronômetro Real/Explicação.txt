Será obtido o Timestamp atual e a cada segundo, será pego um novo  Timestamp que resultará um valor em milisegundos.

    // Your existing code that adds event listeners to the buttons

    btn_iniciar.addEventListener("click",(evt)=>{
        tmpini=Date.now()

    })

    btn_parar.addEventListener("click",(evt)=>{

    })

    btn_zerar.addEventListener("click",(evt)=>{
        tmpini=Date.now()
